# npm 安装

使用 npm 或 yarn 的方式安装。

npm/yarn install/add clw-ui
npm/yarn install/add element-ui

# 快速上手

import CLWUI from 'vxe-table'
import 'clw-ui/lib/clw-ui.css'

import ElementUI from 'element-ui'
import 'element-ui/lib/theme-chalk/index.css'

Vue.use(CLWUI)
Vue.use(ElementUI)
